FoodOrderApp/
│
├── server.js                 # Entry point: Khởi tạo Express, kết nối Middleware & Routes.
├── .env                      # Lưu biến môi trường: DB_PASS, GOOGLE_KEY, VNP_HASH.
├── package.json              # Khai báo thư viện: express, mysql2, axios, multer, dotenv.
│
├── client/                   # FRONTEND (Giao diện người dùng - JS thuần)
│   ├── assets/               # Tài nguyên tĩnh
│   │   ├── css/              # style.css (Dùng chung cho toàn web)
│   │   └── img/              # Ảnh món ăn & Ảnh giấy phép (Multer sẽ lưu vào đây)
│   ├── scripts/              # Logic JavaScript Client-side
│   │   ├── api.js            # Trục xương sống: Chứa các hàm fetch() gọi lên Backend.
│   │   ├── maps.js           # Xử lý Geolocation & Google Maps Distance API.
│   │   ├── cart.js           # Quản lý giỏ hàng (LocalStorage), tính tiền, phí ship.
│   │   └── ui-render.js      # Vẽ dữ liệu ra HTML (List món, Bảng thống kê, Đơn hàng).
│   └── views/                # Các trang HTML (Tách theo Role)
│       ├── index.html        # Khách: Xem món, thấy khoảng cách KM & đặt hàng.
│       ├── checkout.html     # Khách: Xác nhận đơn (COD/VNPay) & Thanh toán.
│       ├── admin-verify.html # Admin: Phê duyệt yêu cầu nâng cấp Role của Seller.
│       ├── seller-dash.html  # Seller: Xem Dashboard thống kê doanh thu & đơn hàng.
│       ├── manage-food.html  # Seller: Giao diện CRUD (Thêm/Sửa/Xóa món ăn).
│       └── auth/             # Login, Register, OTP (Làm ở bước cuối).
│
└── server/                   # BACKEND (Nghiệp vụ - Node.js)
    ├── config/               # db.js: Kết nối MySQL Connection Pool.
    ├── middleware/           # auth.js: Chốt chặn bảo mật, kiểm tra quyền Admin/Seller.
    ├── routes/               # Định nghĩa các cổng API
    │   ├── foodRoutes.js     # API lấy danh sách món, tính KM, CRUD món cho Seller.
    │   ├── orderRoutes.js    # API đặt hàng (User) & Quản lý đơn (Seller).
    │   ├── adminRoutes.js    # API phê duyệt Seller (Admin).
    │   └── paymentRoutes.js  # (Làm cuối) API tạo URL VNPay & IPN.
    ├── controllers/          # XỬ LÝ LOGIC (Brains of the app)
    │   ├── foodController.js # Tính KM, xử lý CRUD món ăn cho Seller.
    │   ├── orderController.js# Xử lý User đặt hàng, Seller xác nhận đơn.
    │   ├── adminController.js# Logic nâng cấp Role từ Customer sang Seller.
    │   └── statsController.js# Logic SQL (SUM, COUNT) để làm thống kê doanh thu.
    ├── services/             # Dịch vụ bên thứ 3
    │   ├── mapsService.js    # Gọi Google Maps API để lấy khoảng cách thực tế.
    │   └── vnpayService.js   # (Làm cuối) Xử lý Hash & Checksum cho VNPay.
    └── models/               # Tầng dữ liệu (Database Layer)
        └── queries.js        # Nơi tập trung toàn bộ câu lệnh SQL thuần.
